C 함수 반환값에 대한 생각

2020-04-07

C언어의 반환값

주로 사용되는 반환값 종류

  • 결과
  • 오류코드

이 두 가지 중 우선순위를 두자면 오류코드를 우선적으로 반환해야 한다. 아래 두 예시를 보며 그 이유를 느낄 수 있다.

out_error = func(&out_result);
if (out_error == error)
    error_handling();
/*또는*/
if ((out_error = func(&out_result)) == error)
    error_handling();

out_result = func(&out_error);
if (out_error == error)
    error_handling();

호출자가 오류를 확인하기에는 오류를 반환해주는 경우가 더 직관적이다. 호출자가 오류를 확인하지 않은 경우 더욱 오류를 반환해주는 쪽이 편리하다.

/* 반환값 버려짐 */func(&out_result);
/* out_result를 사용하는 코드 */

out_result = func(&out_error); /* 쓸모없는 out_error 변수 필요 */
/* out_result를 사용하는 코드 */

당연한 이야기이지만 오류를 반환해주는 쪽이 어셈블리가 조금 짧다. 위쪽이 오류를 반환하는 함수의 어셈블리, 아래쪽이 결과를 반환하는 함수의 어셈블리.

main: # @main
  push rbp
  mov rbp, rsp
  sub rsp, 16
  mov dword ptr [rbp - 4], 0
  lea rdi, [rbp - 8]
  call func
  cmp eax, 0
  jne .LBB1_2
  movabs rdi, offset .L.str
  mov al, 0
  call printf
  jmp .LBB1_3

main: # @main
  push rbp
  mov rbp, rsp
  sub rsp, 16
  mov dword ptr [rbp - 4], 0
  lea rdi, [rbp - 12]
  call func
  mov dword ptr [rbp - 8], eax
  cmp dword ptr [rbp - 12], 0
  jne .LBB1_2
  movabs rdi, offset .L.str
  mov al, 0
  call printf
  jmp .LBB1_3

반환값이 없을 것 같은 함수에도 그럴듯한 반환값을 주는 것이 좋다. 논리 연산자를 이용한 함수 연결을 가능하게 하여 때로 코드를 더 간결하게 유지해준다. strcpy처럼 반환값이 필요하지 않을 것 같은 함수도 반환값이 존재한다.

void     foo(char *dst, char *src)
{
    if (dst != NULL)
    {
        strcpy(dst, src);
        if (strlen(dst) > 1)
            *dst = '*';
    }
}

void    bar(char *dst, char *src)
{
    if (dst != NULL && strcpy(dst, src) && strlen(dst) > 1)
        *dst = '*';
}

아래는 ‘x86-64 clang10.0.0’ 컴파일러로 컴파일한 결과이다.

foo: # @foo
  push rbp
  mov rbp, rsp
  sub rsp, 32
  mov qword ptr [rbp - 8], rdi
  mov qword ptr [rbp - 16], rsi
  cmp qword ptr [rbp - 8], 0
  je .LBB0_4
  mov rdi, qword ptr [rbp - 8]
  mov rsi, qword ptr [rbp - 16]
  call strcpy
  mov rdi, qword ptr [rbp - 8]
  mov qword ptr [rbp - 24], rax
  call strlen
  cmp rax, 1
  jbe .LBB0_3
  mov rax, qword ptr [rbp - 8]
  mov byte ptr [rax], 42
.LBB0_3:
  jmp .LBB0_4
.LBB0_4:
  add rsp, 32
  pop rbp
  ret

bar: # @bar
  push rbp
  mov rbp, rsp
  sub rsp, 16
  mov qword ptr [rbp - 8], rdi
  mov qword ptr [rbp - 16], rsi
  cmp qword ptr [rbp - 8], 0
  je .LBB1_4
  mov rdi, qword ptr [rbp - 8]
  mov rsi, qword ptr [rbp - 16]
  call strcpy
  cmp rax, 0
  je .LBB1_4
  mov rdi, qword ptr [rbp - 8]
  call strlen
  cmp rax, 1
  jbe .LBB1_4
  mov rax, qword ptr [rbp - 8]
  mov byte ptr [rax], 42
.LBB1_4:
  add rsp, 16
  pop rbp
  ret

이처럼 다양한 방법으로 활용할 수 있기에 무엇을 반환해 줄지를 정하는 것도 중요하다.
